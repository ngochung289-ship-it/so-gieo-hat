Copy URL deploy duoc, dan vao app HTML o bien SCRIPT_URL // ======================================================== const SHEET_NAME_LOG = 'Check-in Log'; const SHEET_NAME_KIM = 'So Kim Cuong'; const SHEET_NAME_DCI = 'Lan Toa DCI'; const SHEET_NAME_COURSES = 'Khoa Hoc'; const SHEET_NAME_BANNER = 'Banner'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.type === 'checkin') { saveCheckin(ss, data); } else if (data.type === 'kim') { saveKim(ss, data); } else if (data.type === 'dci') { saveDCI(ss, data); } return ContentService .createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const type = e.parameter.type || 'checkin'; if (type === 'ping') { return ContentService .createTextOutput(JSON.stringify({ success: true, message: 'pong' })) .setMimeType(ContentService.MimeType.JSON); } else if (type === 'checkin') { return getCheckins(ss); } else if (type === 'kim') { return getKims(ss); } else if (type === 'courses') { return getCourses(ss); } else if (type === 'banner') { return getBanner(ss); } } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function getOrCreateSheet(ss, name, headers) { let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length) .setBackground('#534AB7') .setFontColor('#ffffff') .setFontWeight('bold'); sheet.setFrozenRows(1); } return sheet; } function saveCheckin(ss, data) { const headers = [ 'Ngay', 'Muc tieu', 'Nguoi cung mong muon', 'Nguoi co on', 'Nguoi giup nhieu nguoi', 'Truong hop khan cap', 'Gieo truc tiep', 'Gieo gian tiep', 'Day 4 Buoc', 'Buoc 3 - Hanh dong', '4a - Biet on hom nay', '4b - Nguong mo', '4c - Tang buddy', '4d - Biet on 4 buoc', 'B1 - Cay but', 'B2 - An han', 'B3 - Cam ket', 'B4 - Can bang' ]; const sheet = getOrCreateSheet(ss, SHEET_NAME_LOG, headers); // Kiem tra xem ngay nay da co chua — neu co thi merge & cap nhat const data_range = sheet.getDataRange().getValues(); let rowIndex = -1; let existing = null; for (let i = 1; i < data_range.length; i++) { if (data_range[i][0] === data.date) { rowIndex = i + 1; existing = data_range[i]; break; } } // Neu da co dong cu, giu lai gia tri cu cho field nao khong duoc gui len lan nay function pick(newVal, existingIdx) { const hasNew = newVal !== undefined && newVal !== null && newVal !== '' && newVal !== -1; if (hasNew) return newVal; return existing ? existing[existingIdx] : ''; } const row = [ data.date, pick(data.goal, 1), pick(data.landBuddy, 2), pick(data.landOn, 3), pick(data.landNhieu, 4), pick(data.landKhan, 5), pick(data.tructiep, 6), pick(data.giantiep, 7), pick(data.day4buoc, 8), pick(data.b3, 9), pick(data.b4a, 10), pick(data.b4b, 11), pick(data.b4c, 12), pick(data.b4d, 13), pick(data.bs1, 14), pick(data.bs2, 15), pick(data.bs3, 16), pick(data.bs4, 17) ]; if (rowIndex > 0) { sheet.getRange(rowIndex, 1, 1, row.length).setValues([row]); } else { sheet.appendRow(row); // To mau xen ke const lastRow = sheet.getLastRow(); if (lastRow % 2 === 0) { sheet.getRange(lastRow, 1, 1, headers.length).setBackground('#F5F3FF'); } } // Auto resize columns sheet.autoResizeColumns(1, headers.length); } function saveKim(ss, data) { const headers = [ 'Ngay', 'Loai hat giong', 'Manh dat mau mo', 'So tien (VND)', 'Chi tiet', 'Muc tieu', 'ID' ]; const sheet = getOrCreateSheet(ss, SHEET_NAME_KIM, headers); const row = [ data.date, data.type || '', data.land || '', data.amt ? parseInt(data.amt) : '', data.detail || '', data.mt || '', data.id || Date.now() ]; sheet.appendRow(row); const lastRow = sheet.getLastRow(); if (lastRow % 2 === 0) { sheet.getRange(lastRow, 1, 1, headers.length).setBackground('#FFF8E8'); } sheet.autoResizeColumns(1, headers.length); } function saveDCI(ss, data) { const headers = [ 'Ngay', 'Chuong trinh', 'Doi tuong', 'Khu vuc', 'So nguoi', 'Chi tiet' ]; const sheet = getOrCreateSheet(ss, SHEET_NAME_DCI, headers); const row = [ data.date, data.program || '', data.audience || '', data.area || '', data.count ? parseInt(data.count) : '', data.detail || '' ]; sheet.appendRow(row); sheet.autoResizeColumns(1, headers.length); } function getCheckins(ss) { const sheet = ss.getSheetByName(SHEET_NAME_LOG); if (!sheet) { return ContentService .createTextOutput(JSON.stringify({ success: true, data: [] })) .setMimeType(ContentService.MimeType.JSON); } const values = sheet.getDataRange().getValues(); const headers = values[0]; const rows = values.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify({ success: true, data: rows })) .setMimeType(ContentService.MimeType.JSON); } function getKims(ss) { const sheet = ss.getSheetByName(SHEET_NAME_KIM); if (!sheet) { return ContentService .createTextOutput(JSON.stringify({ success: true, data: [] })) .setMimeType(ContentService.MimeType.JSON); } const values = sheet.getDataRange().getValues(); const headers = values[0]; const rows = values.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify({ success: true, data: rows })) .setMimeType(ContentService.MimeType.JSON); } function getCourses(ss) { // Tu dong tao sheet neu chua co, voi headers chuan const headers = [ 'TenKhoaHoc', 'MoTa', 'GiaoVien', 'NgayKhaiGiang', 'HocPhi', 'LinkDangKy', 'AnhURL', 'Tag', 'HienThi' ]; const sheet = getOrCreateSheet(ss, SHEET_NAME_COURSES, headers); const values = sheet.getDataRange().getValues(); if (values.length <= 1) { return ContentService .createTextOutput(JSON.stringify({ success: true, data: [] })) .setMimeType(ContentService.MimeType.JSON); } const hdrs = values[0]; const rows = values.slice(1) .filter(row => row[hdrs.indexOf('HienThi')] !== 'No' && row[hdrs.indexOf('HienThi')] !== false && row[0] !== '') .map(row => { const obj = {}; hdrs.forEach((h, i) => obj[h] = row[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify({ success: true, data: rows })) .setMimeType(ContentService.MimeType.JSON); } function getBanner(ss) { // Tu dong tao sheet neu chua co, voi headers chuan const headers = ['AnhURL', 'GhiChu']; const sheet = getOrCreateSheet(ss, SHEET_NAME_BANNER, headers); const values = sheet.getDataRange().getValues(); if (values.length <= 1) { return ContentService .createTextOutput(JSON.stringify({ success: true, data: [] })) .setMimeType(ContentService.MimeType.JSON); } const hdrs = values[0]; const rows = values.slice(1) .filter(row => row[hdrs.indexOf('AnhURL')] !== '') .map(row => { const obj = {}; hdrs.forEach((h, i) => obj[h] = row[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify({ success: true, data: rows })) .setMimeType(ContentService.MimeType.JSON); }
